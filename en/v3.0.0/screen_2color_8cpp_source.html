86/// @brief Build a Color from its RGB representation.
87/// https://en.wikipedia.org/wiki/RGB_color_model
88///
89/// @param red The quantity of red [0,255]
90/// @param green The quantity of green [0,255]
91/// @param blue The quantity of blue [0,255]
92/// @ingroup screen
93Color::Color(uint8_t red, uint8_t green, uint8_t blue)
94 : type_(ColorType::TrueColor), red_(red), green_(green), blue_(blue) {
95 if (Terminal::ColorSupport() == Terminal::Color::TrueColor) {
96 return;
97 }
98
99 // Find the closest Color from the database:
100 const int max_distance = 256 * 256 * 3;
101 int closest = max_distance;
102 int best = 0;
103 const int database_begin = 16;
104 const int database_end = 256;
105 for (int i = database_begin; i < database_end; ++i) {
106 ColorInfo color_info = GetColorInfo(Color::Palette256(i));
107 int dr = color_info.red - red;
108 int dg = color_info.green - green;
109 int db = color_info.blue - blue;
110 int dist = dr * dr + dg * dg + db * db;
111 if (closest > dist) {
112 closest = dist;
113 best = i;
114 }
115 }
116
117 if (Terminal::ColorSupport() == Terminal::Color::Palette256) {
118 type_ = ColorType::Palette256;
119 red_ = best;
120 } else {
121 type_ = ColorType::Palette16;
122 red_ = GetColorInfo(Color::Palette256(best)).index_16;
123 }
124}

138/// @brief Build a Color from its HSV representation.
139/// https://en.wikipedia.org/wiki/HSL_and_HSV
140///
141/// @param h The hue of the color [0,255]
142/// @param s The "colorfulness" [0,255].
143/// @param v The "Lightness" [0,255]
144/// @ingroup screen
145// static
146Color Color::HSV(uint8_t h, uint8_t s, uint8_t v) {
147 if (s == 0) {
148 return Color(v, v, v);
149 }
150
151 uint8_t region = h / 43; // NOLINT
152 uint8_t remainder = (h - (region * 43)) * 6; // NOLINT
153 uint8_t p = (v * (255 - s)) >> 8; // NOLINT
154 uint8_t q = (v * (255 - ((s * remainder) >> 8))) >> 8; // NOLINT
155 uint8_t t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8; // NOLINT
156
157 // clang-format off
158 switch (region) { // NOLINT
159 case 0: return Color(v,t,p); // NOLINT
160 case 1: return Color(q,v,p); // NOLINT
161 case 2: return Color(p,v,t); // NOLINT
162 case 3: return Color(p,q,v); // NOLINT
163 case 4: return Color(t,p,v); // NOLINT
164 case 5: return Color(v,p,q); // NOLINT
165 } // NOLINT
166 // clang-format on
167
168 return Color(0, 0, 0);
169}
