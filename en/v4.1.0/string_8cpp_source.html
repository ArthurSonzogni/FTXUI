1481// Find a codepoint inside a sorted list of Interval.
1482template <size_t N>
1483bool Bisearch(uint32_t ucs, const std::array<Interval, N> table) {
1484 if (ucs < table.front().first || ucs > table.back().last) { // NOLINT
1485 return false;
1486 }
1487
1488 int min = 0;
1489 int max = N - 1;
1490 while (max >= min) {
1491 const int mid = (min + max) / 2;
1492 if (ucs > table[mid].last) { // NOLINT
1493 min = mid + 1;
1494 } else if (ucs < table[mid].first) { // NOLINT
1495 max = mid - 1;
1496 } else {
1497 return true;
1498 }
1499 }
1500
1501 return false;
1502}

1568// From UTF8 encoded string |input|, eat in between 1 and 4 byte representing
1569// one codepoint. Put the codepoint into |ucs|. Start at |start| and update
1570// |end| to represent the beginning of the next byte to eat for consecutive
1571// executions.
1572bool EatCodePoint(const std::string& input,
1573 size_t start,
1574 size_t* end,
1575 uint32_t* ucs) {
1576 if (start >= input.size()) {
1577 *end = start + 1;
1578 return false;
1579 }
1580 const uint8_t C0 = input[start];
1581
1582 // 1 byte string.
1583 if ((C0 & 0b1000'0000) == 0b0000'0000) { // NOLINT
1584 *ucs = C0 & 0b0111'1111; // NOLINT
1585 *end = start + 1;
1586 return true;
1587 }
1588
1589 // 2 byte string.
1590 if ((C0 & 0b1110'0000) == 0b1100'0000 && // NOLINT
1591 start + 1 < input.size()) {
1592 const uint8_t C1 = input[start + 1];
1593 *ucs = 0;
1594 *ucs += C0 & 0b0001'1111; // NOLINT
1595 *ucs <<= 6; // NOLINT
1596 *ucs += C1 & 0b0011'1111; // NOLINT
1597 *end = start + 2;
1598 return true;
1599 }
1600
1601 // 3 byte string.
1602 if ((C0 & 0b1111'0000) == 0b1110'0000 && // NOLINT
1603 start + 2 < input.size()) {
1604 const uint8_t C1 = input[start + 1];
1605 const uint8_t C2 = input[start + 2];
1606 *ucs = 0;
1607 *ucs += C0 & 0b0000'1111; // NOLINT
1608 *ucs <<= 6; // NOLINT
1609 *ucs += C1 & 0b0011'1111; // NOLINT
1610 *ucs <<= 6; // NOLINT
1611 *ucs += C2 & 0b0011'1111; // NOLINT
1612 *end = start + 3;
1613 return true;
1614 }
1615
1616 // 4 byte string.
1617 if ((C0 & 0b1111'1000) == 0b1111'0000 && // NOLINT
1618 start + 3 < input.size()) {
1619 const uint8_t C1 = input[start + 1];
1620 const uint8_t C2 = input[start + 2];
1621 const uint8_t C3 = input[start + 3];
1622 *ucs = 0;
1623 *ucs += C0 & 0b0000'0111; // NOLINT
1624 *ucs <<= 6; // NOLINT
1625 *ucs += C1 & 0b0011'1111; // NOLINT
1626 *ucs <<= 6; // NOLINT
1627 *ucs += C2 & 0b0011'1111; // NOLINT
1628 *ucs <<= 6; // NOLINT
1629 *ucs += C3 & 0b0011'1111; // NOLINT
1630 *end = start + 4;
1631 return true;
1632 }
1633
1634 *end = start + 1;
1635 return false;
1636}

1638// From UTF16 encoded string |input|, eat in between 1 and 4 byte representing
1639// one codepoint. Put the codepoint into |ucs|. Start at |start| and update
1640// |end| to represent the beginning of the next byte to eat for consecutive
1641// executions.
1642bool EatCodePoint(const std::wstring& input,
1643 size_t start,
1644 size_t* end,
1645 uint32_t* ucs) {
1646 if (start >= input.size()) {
1647 *end = start + 1;
1648 return false;
1649 }
1650
1651 // On linux wstring uses the UTF32 encoding:
1652 if constexpr (sizeof(wchar_t) == 4) {
1653 *ucs = input[start]; // NOLINT
1654 *end = start + 1;
1655 return true;
1656 }
1657
1658 // On windows, wstring uses the UTF16 encoding:
1659 int32_t C0 = input[start]; // NOLINT
1660
1661 // 1 word size:
1662 if (C0 < 0xd800 || C0 >= 0xdc00) { // NOLINT
1663 *ucs = C0;
1664 *end = start + 1;
1665 return true;
1666 }
1667
1668 // 2 word size:
1669 if (start + 1 >= input.size()) {
1670 *end = start + 2;
1671 return false;
1672 }
1673
1674 int32_t C1 = input[start + 1]; // NOLINT
1675 *ucs = ((C0 & 0x3ff) << 10) + (C1 & 0x3ff) + 0x10000; // NOLINT
1676 *end = start + 2;
1677 return true;
1678}

1772 return 0;

1781 start = end;
1782 continue;
1783 }

1786 // requested, return its start position immediately.
1787 if (glyph_index == 0) {
1788 return static_cast<int>(start);
1789 }
1790
1791 // Otherwise, skip this glyph and iterate:
1792 glyph_index--;
1793 start = end;
1794 }
1795 return static_cast<int>(input.size());

1898/// Convert a UTF8 std::string into a std::wstring.
1899std::string to_string(const std::wstring& s) {
1900 std::string out;
1901
1902 size_t i = 0;
1903 uint32_t codepoint = 0;
1904 while (EatCodePoint(s, i, &i, &codepoint)) {
1905 // Code point <-> UTF-8 conversion
1906 //
1907 // ┏━━━━━━━━┳━━━━━━━━┳━━━━━━━━┳━━━━━━━━┓
1908 // ┃Byte 1 ┃Byte 2 ┃Byte 3 ┃Byte 4 ┃
1909 // ┡━━━━━━━━╇━━━━━━━━╇━━━━━━━━╇━━━━━━━━┩
1910 // │0xxxxxxx│ │ │ │
1911 // ├────────┼────────┼────────┼────────┤
1912 // │110xxxxx│10xxxxxx│ │ │
1913 // ├────────┼────────┼────────┼────────┤
1914 // │1110xxxx│10xxxxxx│10xxxxxx│ │
1915 // ├────────┼────────┼────────┼────────┤
1916 // │11110xxx│10xxxxxx│10xxxxxx│10xxxxxx│
1917 // └────────┴────────┴────────┴────────┘
1918
1919 // 1 byte UTF8
1920 if (codepoint <= 0b000'0000'0111'1111) { // NOLINT
1921 const uint8_t p1 = codepoint;
1922 out.push_back(p1); // NOLINT
1923 continue;
1924 }
1925
1926 // 2 bytes UTF8
1927 if (codepoint <= 0b000'0111'1111'1111) { // NOLINT
