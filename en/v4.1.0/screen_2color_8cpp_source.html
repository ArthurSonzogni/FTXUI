87/// @brief Build a Color from its RGB representation.
88/// https://en.wikipedia.org/wiki/RGB_color_model
89///
90/// @param red The quantity of red [0,255]
91/// @param green The quantity of green [0,255]
92/// @param blue The quantity of blue [0,255]
93/// @ingroup screen
94Color::Color(uint8_t red, uint8_t green, uint8_t blue)
95 : type_(ColorType::TrueColor), red_(red), green_(green), blue_(blue) {
96 if (Terminal::ColorSupport() == Terminal::Color::TrueColor) {
97 return;
98 }
99
100 // Find the closest Color from the database:
101 const int max_distance = 256 * 256 * 3;
102 int closest = max_distance;
103 int best = 0;
104 const int database_begin = 16;
105 const int database_end = 256;
106 for (int i = database_begin; i < database_end; ++i) {
107 const ColorInfo color_info = GetColorInfo(Color::Palette256(i));
108 const int dr = color_info.red - red;
109 const int dg = color_info.green - green;
110 const int db = color_info.blue - blue;
111 const int dist = dr * dr + dg * dg + db * db;
112 if (closest > dist) {
113 closest = dist;
114 best = i;
115 }
116 }
117
118 if (Terminal::ColorSupport() == Terminal::Color::Palette256) {
119 type_ = ColorType::Palette256;
120 red_ = best;
121 } else {
122 type_ = ColorType::Palette16;
123 red_ = GetColorInfo(Color::Palette256(best)).index_16;
124 }
125}

139/// @brief Build a Color from its HSV representation.
140/// https://en.wikipedia.org/wiki/HSL_and_HSV
141///
142/// @param h The hue of the color [0,255]
143/// @param s The "colorfulness" [0,255].
144/// @param v The "Lightness" [0,255]
145/// @ingroup screen
146// static
147Color Color::HSV(uint8_t h, uint8_t s, uint8_t v) {
148 if (s == 0) {
149 return {0, 0, 0};
150 }
151
152 uint8_t region = h / 43; // NOLINT
153 uint8_t remainder = (h - (region * 43)) * 6; // NOLINT
154 uint8_t p = (v * (255 - s)) >> 8; // NOLINT
155 uint8_t q = (v * (255 - ((s * remainder) >> 8))) >> 8; // NOLINT
156 uint8_t t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8; // NOLINT
157
158 // clang-format off
159 switch (region) { // NOLINT
160 case 0: return Color(v,t,p); // NOLINT
161 case 1: return Color(q,v,p); // NOLINT
162 case 2: return Color(p,v,t); // NOLINT
163 case 3: return Color(p,q,v); // NOLINT
164 case 4: return Color(t,p,v); // NOLINT
165 case 5: return Color(v,p,q); // NOLINT
166 } // NOLINT
167 // clang-format on
168 return {0, 0, 0};
169}
