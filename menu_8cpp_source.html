239  bool OnEvent(Event event) override {
240  Clamp();
241  if (!CaptureMouse(event)) {
242  return false;
243  }
244 
245  if (event.is_mouse()) {
246  return OnMouseEvent(event);
247  }
248 
249  if (Focused()) {
250  const int old_selected = selected();
251  if (event == Event::ArrowUp || event == Event::Character('k')) {
252  OnUp();
253  }
254  if (event == Event::ArrowDown || event == Event::Character('j')) {
255  OnDown();
256  }
257  if (event == Event::ArrowLeft || event == Event::Character('h')) {
258  OnLeft();
259  }
260  if (event == Event::ArrowRight || event == Event::Character('l')) {
261  OnRight();
262  }
263  if (event == Event::PageUp) {
264  selected() -= box_.y_max - box_.y_min;
265  }
266  if (event == Event::PageDown) {
267  selected() += box_.y_max - box_.y_min;
268  }
269  if (event == Event::Home) {
270  selected() = 0;
271  }
272  if (event == Event::End) {
273  selected() = size() - 1;
274  }
275  if (event == Event::Tab && size()) {
276  selected() = (selected() + 1) % size();
277  }
278  if (event == Event::TabReverse && size()) {
279  selected() = (selected() + size() - 1) % size();
280  }
281 
282  selected() = util::clamp(selected(), 0, size() - 1);
283 
284  if (selected() != old_selected) {
285  focused_entry() = selected();
286  SelectedTakeFocus();
287  OnChange();
288  return true;
289  }
290  }
291 
292  if (event == Event::Return) {
293  OnEnter();
294  return true;
295  }
296 
297  return false;
298  }

300  bool OnMouseEvent(Event event) {
301  if (event.mouse().button == Mouse::WheelDown ||
302  event.mouse().button == Mouse::WheelUp) {
303  return OnMouseWheel(event);
304  }
305 
306  if (event.mouse().button != Mouse::None &&
307  event.mouse().button != Mouse::Left) {
308  return false;
309  }
310  if (!CaptureMouse(event)) {
311  return false;
312  }
313  for (int i = 0; i < size(); ++i) {
314  if (!boxes_[i].Contain(event.mouse().x, event.mouse().y)) {
315  continue;
316  }
317 
318  TakeFocus();
319  focused_entry() = i;
320 
321  if (event.mouse().button == Mouse::Left &&
322  event.mouse().motion == Mouse::Pressed) {
323  if (selected() != i) {
324  selected() = i;
325  selected_previous_ = selected();
326  OnChange();
327  }
328  return true;
329  }
330  }
331  return false;
332  }
