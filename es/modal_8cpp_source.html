54// Decora un componente. Añade una ventana |modal| encima de él. Se muestra una sobre
55// la otra cuando |show_modal| es verdadero.
56/// @ingroup component
57// NOLINTNEXTLINE
58ComponentDecorator Modal(Component modal, const bool* show_modal) {
