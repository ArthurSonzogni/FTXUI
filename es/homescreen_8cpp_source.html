37 auto my_graph = [&shift](int width, int height) {
38 std::vector<int> output(width);
39 for (int i = 0; i < width; ++i) {
40 float v = 0.5f;
41 v += 0.1f * sin((i + shift) * 0.1f);
42 v += 0.2f * sin((i + shift + 10) * 0.15f);
43 v += 0.1f * sin((i + shift) * 0.03f);
44 v *= height;
45 output[i] = (int)v;
46 }
47 return output;
48 };
