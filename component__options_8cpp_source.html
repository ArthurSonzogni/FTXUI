76 MenuOption MenuOption::Horizontal() {
77  MenuOption option;
78  option.direction = Direction::Right;
79  option.entries_option.transform = [](const EntryState& state) {
80  Element e = text(state.label);
81  if (state.focused) {
82  e |= inverted;

84  if (state.active) {
85  e |= bold;
86  }

89  }
90  return e;
91  };
