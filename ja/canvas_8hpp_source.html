134 struct XY {
135 int x;
136 int y;
137 bool operator==(const XY& other) const {
138 return x == other.x && y == other.y;
139 }
140 };
141
142 struct XYHash {
143 size_t operator()(const XY& xy) const {
144 constexpr size_t shift = 1024;
145 return size_t(xy.x) * shift + size_t(xy.y);
146 }
147 };
