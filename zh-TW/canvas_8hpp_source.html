133 struct XY {
134 int x;
135 int y;

137 return x == other.x && y == other.y;
138 }
139 };

141 struct XYHash {
142 size_t operator()(const XY& xy) const {
143 constexpr size_t shift = 1024;
144 return size_t(xy.x) * shift + size_t(xy.y);
