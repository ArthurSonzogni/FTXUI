19MenuEntryOption Colored(ftxui::Color c) {
20 MenuEntryOption option;
21 option.transform = [c](EntryState state) {
22 state.label = (state.active ? "> " : " ") + state.label;
23 Element e = text(state.label) | color(c);
24 if (state.focused) {
25 e = e | inverted;
26 }
27 if (state.active) {
28 e = e | bold;
29 }
30 return e;
31 };
32 return option;
33}
